# ============ TAB 3: CURRENT OCCUPANCY (Updated) ============
with tab3:
    st.header("👥 Persons Currently Inside")
    
    if st.button("🔄 Refresh", use_container_width=True):
        st.rerun()
    
    try:
        status = requests.get(f"{API_URL}/vehicle_status", timeout=2).json()
        
        # Summary cards
        col1, col2, col3, col4 = st.columns(4)
        with col1:
            st.metric("🚪 Total Inside", status.get("current_occupancy", 0))
        with col2:
            st.metric("👨 Males", status.get("male_count", 0))
        with col3:
            st.metric("👩 Females", status.get("female_count", 0))
        with col4:
            unknown_count = status.get("unknown_person_count", 0)
            st.metric("❓ Unknown Persons", unknown_count)
        
        # Show total unknown detected separately
        total_unknown = status.get("total_unknown_detected", 0)
        if total_unknown > 0:
            st.warning(f"⚠️ Total Unknown Persons Detected (Alerted): {total_unknown}")
        
        st.markdown("---")
        
        # Detailed list of active passengers
        active_passengers = status.get("active_passengers", [])
        if active_passengers:
            df_data = []
            for p in active_passengers:
                df_data.append({
                    "Name": p.get("passenger_name"),
                    "ID": p.get("passenger_id"),
                    "Gender": p.get("gender", "N/A"),
                    "Type": p.get("passenger_type"),
                    "Entry Time": p.get("entry_time"),
                    "Entry Date": p.get("entry_date"),
                    "Entry Location": p.get("entry_location"),
                    "Entry GPS": p.get("entry_gps"),
                    "Status": "🟢 IN VEHICLE"
                })
            df = pd.DataFrame(df_data)
            st.dataframe(df, use_container_width=True)
            
            # Gender distribution chart
            st.subheader("Gender Distribution (Inside)")
            gender_data = pd.DataFrame({
                "Category": ["Male", "Female", "Unknown Person"],
                "Count": [
                    status.get("male_count", 0), 
                    status.get("female_count", 0), 
                    status.get("unknown_person_count", 0)
                ]
            })
            st.bar_chart(gender_data.set_index("Category"))
        else:
            st.info("No persons currently inside")
            
            # Show unknown detection history
            unknown_count = status.get("total_unknown_detected", 0)
            if unknown_count > 0:
                st.warning(f"⚠️ {unknown_count} unknown person(s) have been detected and alerted previously")
    
    except Exception as e:
        st.error(f"Error: {e}")
